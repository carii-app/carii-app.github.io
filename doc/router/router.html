final router = GoRouter(
  routes: [
    GoRoute(
      path: start,
      builder: (context, state) => const SplashView(),
    ),
    GoRoute(
      path: loginUrl,
      builder: (context, state) => const LoginView(),
    ),
    GoRoute(
      path: loginConfirmedUrl,
      builder: (context, state) => const LoginConfirmedView(),
    ),
    GoRoute(
      path: homeUrl,
      builder: (context, state) => const HomeView(),
    ),
  ],
  redirect: (context, state) async {
    // if the user is not logged in, they need to login
    final loggedIn = supabase.auth.currentUser != null;
    final loggingIn = state.location == loginUrl;
    if (!loggedIn) return loggingIn ? null : loginUrl;

    // if the user is logged in but still on the login page, send them to
    // the confirmed login page
    if (loggingIn) return loginConfirmedUrl;

    //if user was never logged in send them to splash (/)
    if (!await getWasAlreadyLoggedIn()) {
      return start;
    }

    //TO-DO: if he was alreday logged in send him to home (/home)
    //TO-DO: keep that in sync with SplashView and stuff

    // no need to redirect at all
    return null;
  },
);
