LoginModel() : super(const LoginState()) {
  getWasAlreadyLoggedIn().then((isNotFirstLogin) => state = state.copyWith(
        isFirstLogin: !isNotFirstLogin,
        email: supabase.auth.currentUser?.email,
        dirty: false,
      ));
  // final authSub =
  supabase.auth.onAuthStateChange.listen((data) {
    final event = data.event;
    final session = data.session;
    final user = session?.user;
    if (event == AuthChangeEvent.signedIn) {
      state = state.copyWith(
        isFirstLogin: false,
        email: user?.email,
        dirty: false,
      );
      _success(session!, user!); //this might be called multiple times
    } else if (event == AuthChangeEvent.signedOut) {
      _logoutSuccess(); //this might be called multiple times
    }
  });
  // checkLoginPoll();
}
