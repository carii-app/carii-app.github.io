List<Widget> loginUI(MainTextField emailField, LoginState loginState,
    BuildContext context, WidgetRef ref) {
  return [
    emailField,
    if (!loginState.useOTPFlow) ...[
      const SizedBox(height: 8),
      passwordField(
        loginState,
        context,
        ref,
        name: AppLocalizations.of(context)!.password,
        onChanged: (value) {
          ref.read(loginProvider.notifier).passwordChanged(value);
        },
      ),
      if (loginState.register) const SizedBox(height: 8),
      if (loginState.register)
        passwordField(
          loginState,
          context,
          ref,
          name: AppLocalizations.of(context)!.repeatPassword,
          onChanged: (value) {
            ref.read(loginProvider.notifier).repeatPasswordChanged(value);
          },
        ),
    ],
    const SizedBox(height: 32),
    Hero(
      tag: 'mainActionButton-0',
      child: MainFullSizeTextButtonT(
        loginState.useOTPFlow
            ? AppLocalizations.of(context)!.registerOrLoginButton
            : loginState.register
                ? AppLocalizations.of(context)!.registerButton
                : AppLocalizations.of(context)!.loginButton,
        onPressed: () {
          loginState.register
              ? ref.read(loginProvider.notifier).registerPressed()
              : ref.read(loginProvider.notifier).loginPressed();
        },
      ),
    ),
  ];
}
